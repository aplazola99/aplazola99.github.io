sentence = "Wow, this is super crazy."
counts = {}
for c in sentence:
    if c in counts:
        counts[c]+=1
    else:
        counts[c]=1
def count_chars(sentence):
    counts = {}
    for c in sentence:
        if c in counts:
            counts[c]+=1
        else:
            counts[c]=1
    for c in counts:
        print(c, "appears", counts[c],"times")
count_chars(strings2)
